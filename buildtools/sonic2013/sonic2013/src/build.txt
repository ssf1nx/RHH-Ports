set -e

# Retry git clone on transient HTTP 500s.
git_clone_retry() {
    local url target attempt
    for arg in "$@"; do
        case "$arg" in
            http*|git@*|ssh://*|git://*) url="$arg" ;;
            -*) ;;
            *) [[ -n "$url" ]] && target="$arg" ;;
        esac
    done
    if [[ -z "$target" && -n "$url" ]]; then
        target="${url##*/}"
        target="${target%.git}"
    fi
    for attempt in 1 2 3 4 5; do
        if git clone "$@"; then
            return 0
        fi
        [[ -n "$target" && -d "$target" ]] && rm -rf "$target"
        local delay=$((attempt * 5))
        echo "git clone failed (attempt $attempt); retrying in ${delay}s..." >&2
        sleep "$delay"
    done
    echo "git clone failed after 5 attempts" >&2
    return 1
}

# Build SDL2 2.32 for link-time only. Device provides libSDL2 at runtime.
git_clone_retry https://github.com/libsdl-org/SDL.git
cd SDL
git checkout release-2.32.0
mkdir -p build-sonic2013 && cd build-sonic2013
cmake .. -DBUILD_SHARED_LIBS=ON
make -j$(nproc)
make install
cd ../..

#### RSDKv4 (Sonic 1/2 2013 Decompilation) ####

git_clone_retry --recursive https://github.com/RSDKModding/RSDKv4-Decompilation.git
cd RSDKv4-Decompilation

# Latest release tag, or HEAD if FORCE_HEAD=true.
git fetch --tags
if [[ "${FORCE_HEAD:-false}" == "true" ]]; then
  echo "FORCE_HEAD=true — building from branch HEAD (nightly)"
else
  git checkout $(git describe --tags `git rev-list --tags --max-count=1`)
fi

# REVISION=3 is the universal binary — REV03's >= macros include earlier
# revisions' code and runtime detects Origins vs mobile data layout.
# RETRO_USE_HW_RENDER=OFF drops GLEW (no libglvnd on device).
# -DRETRO_DISABLE_LOG stubs out PrintLog, which has an upstream sprintf-
# into-self bug that trips stack protection on focal glibc.
# -pthread fixes a theoraplay --as-needed linker error.
cmake -B build \
  -DCMAKE_BUILD_TYPE=Release \
  -DRETRO_REVISION=3 \
  -DRETRO_USE_HW_RENDER=OFF \
  -DRETRO_DISABLE_PLUS=ON \
  -DCMAKE_CXX_FLAGS="-DRETRO_DISABLE_LOG" \
  -DCMAKE_C_FLAGS="-DRETRO_DISABLE_LOG" \
  -DCMAKE_EXE_LINKER_FLAGS="-pthread"

# Fail if the PrintLog overflow warning survives — means RETRO_DISABLE_LOG
# no longer covers the buggy code and we'd ship a stack-smashing binary.
set -o pipefail
cmake --build build --config Release -j$(nproc) 2>&1 | tee build.log
set +o pipefail
if grep -qE "may write a terminating nul past the end.*format-overflow" build.log; then
    echo "ERROR: sprintf-into-self overflow warning still present — RETRO_DISABLE_LOG may not"
    echo "       be stubbing out the buggy code path any more. Shipping would risk a"
    echo "       'stack smashing detected' abort at runtime."
    exit 1
fi

# Stage shipped libs. Names must match DT_NEEDED; cp -L dereferences the
# SONAME symlink to a real file. RSDKv4 only needs libvorbisfile (no
# intro video path like RSDKv3 has).
mkdir -p build/libs
cp -L /usr/lib/aarch64-linux-gnu/libvorbisfile.so.3 ./build/libs/libvorbisfile.so.3

# Verify every NEEDED entry is either staged or device-provided.
bin=$(find build -maxdepth 3 -name RSDKv4 -type f | head -n1)
if [[ -z "$bin" ]]; then
    echo "ERROR: built RSDKv4 binary not found under build/"; exit 1
fi
echo "Binary: $bin"
needed=$(readelf -d "$bin" | awk -F'[][]' '/NEEDED/ {print $2}')
echo "Binary NEEDED:"; echo "$needed"
echo "Staged:"; ls build/libs/

staged_libs="libvorbisfile.so.3"
# ld-linux-aarch64.so.1 is the dynamic loader; always present.
system_libs="libSDL2-2.0.so.0 libGLESv2.so.2 \
             libc.so.6 libm.so.6 libpthread.so.0 libdl.so.2 librt.so.1 \
             libstdc++.so.6 libgcc_s.so.1 \
             ld-linux-aarch64.so.1"

for lib in $staged_libs; do
    if ! echo "$needed" | grep -qx "$lib"; then
        echo "ERROR: staged $lib but binary does not NEED it"; exit 1
    fi
    if [ ! -f "build/libs/$lib" ]; then
        echo "ERROR: $lib missing from build/libs/"; exit 1
    fi
done

while read -r lib; do
    [[ -z "$lib" ]] && continue
    if ! echo " $staged_libs $system_libs " | grep -q " $lib "; then
        echo "ERROR: binary NEEDs $lib but it is neither staged nor on the system allowlist"; exit 1
    fi
done <<< "$needed"

# $bin is already build/RSDKv4 from CMakeLists' default OUTPUT_NAME;
# retrieve-products copies it into each target_dir as-is.

cd ..
