#!/bin/bash
set -e

SRCDIR=$1
DESTDIR=$2

cd "$DESTDIR" || { echo "DESTDIR not found: $DESTDIR"; exit 1; }

SRC_GAME="$SRCDIR/sonic3air/Oxygen/sonic3air"

# Binary
cp "$SRC_GAME/sonic3air_linux" "$DESTDIR/sonic3air_linux"

# Upstream engine data must match the binary's expectations. We ship it as
# a 7z archive to stay under GitHub's 100MB per-file limit — plain zip of
# this 133MB asset tree lands around 127MB; LZMA compression brings it
# comfortably under. The launcher extracts in-place on first run (or on
# port update when the archive reappears).
cp "$SRC_GAME/config.json" "$DESTDIR/config.json"
# Remove any prior archive (including old zip form) before repackaging.
rm -f "$DESTDIR/data.7z" "$DESTDIR/data.zip"
# 7z is pre-installed on GitHub-hosted ubuntu-24.04 runners. -mx=9 is max
# LZMA compression; for this asset tree it produces ~70-80MB vs zip's 127MB.
(cd "$SRC_GAME" && 7z a -t7z -mx=9 "$DESTDIR/data.7z" data >/dev/null)
echo "data.7z size: $(stat -c %s "$DESTDIR/data.7z") bytes"

# sonic3air ships no bundled libs; `libs/` is intentionally absent from the
# shipped payload. If the build.txt verification step starts failing because
# a NEEDED dep isn't on the system allowlist, add a libs/ stage here and
# update registry.json's artifacts list.
#
# Other files in the port dir (cover.png, gameinfo.xml, port.json,
# sonic.gptk, README.md, saves/, doc/, config/mods/) are RHH packaging or
# runtime-generated state. They are intentionally NOT synced — the sync
# step in build_ports.yml only touches paths listed in the registry's
# `artifacts` field, so everything else is preserved as-is.

echo "Packaging complete."
