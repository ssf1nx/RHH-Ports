set -e

# GitHub occasionally returns HTTP 500s during git clone. Retry with backoff
# so transient failures don't abort the whole build. Use this instead of
# `git clone` directly for any remote clone. Derives the target dir from
# the URL basename (same as `git clone` itself) to clean up partial clones
# between attempts.
git_clone_retry() {
    local url target attempt
    for arg in "$@"; do
        case "$arg" in
            http*|git@*|ssh://*|git://*) url="$arg" ;;
            -*) ;;
            *) [[ -n "$url" ]] && target="$arg" ;;
        esac
    done
    if [[ -z "$target" && -n "$url" ]]; then
        target="${url##*/}"
        target="${target%.git}"
    fi
    for attempt in 1 2 3 4 5; do
        if git clone "$@"; then
            return 0
        fi
        [[ -n "$target" && -d "$target" ]] && rm -rf "$target"
        local delay=$((attempt * 5))
        echo "git clone failed (attempt $attempt); retrying in ${delay}s..." >&2
        sleep "$delay"
    done
    echo "git clone failed after 5 attempts" >&2
    return 1
}

# Build SDL2 2.32 from source so sonic3air links against a modern SDL2 at
# build time. We do NOT ship libSDL2-2.0.so.0 — every supported device
# provides one, often with device-specific patches. Shipping ours via
# LD_LIBRARY_PATH would override those patches.
git_clone_retry https://github.com/libsdl-org/SDL.git
cd SDL
git checkout release-2.32.0
mkdir -p build-sonic3air && cd build-sonic3air
cmake .. -DBUILD_SHARED_LIBS=ON
make -j$(nproc)
make install
cd ../..

#### Sonic 3 AIR ####

git_clone_retry --recursive https://github.com/Eukaryot/sonic3air.git
cd sonic3air

# Check out the latest release tag, or branch HEAD if FORCE_HEAD is set.
git fetch --tags
if [[ "${FORCE_HEAD:-false}" == "true" ]]; then
  echo "FORCE_HEAD=true — building from branch HEAD (nightly)"
else
  # sonic3air tags prereleases too (e.g. -preview, -test). Filter to the
  # latest -stable tag so we don't accidentally ship an unstable build.
  latest=$(git tag --sort=-creatordate | grep -E '\-stable$' | head -n1)
  if [[ -z "$latest" ]]; then
    echo "No -stable tag found; falling back to main HEAD"
  else
    echo "Checking out latest stable tag: $latest"
    git checkout "$latest"
  fi
fi

# Submodule init can also hit transient HTTP 500s from GitHub — retry.
for attempt in 1 2 3 4 5; do
  if git submodule update --init --recursive; then break; fi
  echo "submodule update failed (attempt $attempt); retrying in $((attempt * 5))s..." >&2
  sleep $((attempt * 5))
  [[ $attempt -eq 5 ]] && { echo "submodule update failed after 5 attempts"; exit 1; }
done

mkdir -p Oxygen/sonic3air/build/_cmake/build
cd Oxygen/sonic3air/build/_cmake/build
# USE_GLES=ON links GLESv2 instead of desktop OpenGL (libGLX.so.0 +
# libOpenGL.so.0 from libglvnd). PortMaster devices provide libGLESv2.so.2
# but not libglvnd, so desktop-GL builds fail at runtime. Also defines
# RMX_LINUX_ENFORCE_GLES2 which fixes rmxmedia's GL calls for embedded.
cmake .. -DCMAKE_BUILD_TYPE=Release -DUSE_GLES=ON
make -j$(nproc)
cd ../../../../..

# Run the binary with -pack to package the raw data/ tree into the compact
# .bin files upstream's Linux release ships (gamedata.bin, audiodata.bin,
# audioremaster.bin, enginedata.bin, scripts.bin + metadata.json). This
# shrinks the port payload from ~330MB raw to ~220MB packaged, and matches
# the layout the engine expects at runtime. PackageBuilder writes the bins
# to cwd; metadata.json is written into data/ directly.
cd Oxygen/sonic3air
./sonic3air_linux -pack
# Rebuild data/ containing only the packaged outputs (+ metadata.json,
# + icon.png which is part of the canonical release layout).
mkdir -p data.new
mv gamedata.bin audiodata.bin audioremaster.bin enginedata.bin scripts.bin data.new/
cp data/metadata.json data.new/metadata.json
[[ -f data/icon.png ]] && cp data/icon.png data.new/icon.png
rm -rf data
mv data.new data
echo "Packaged data/ contents:"; ls -l data/
cd ../..

# sonic3air ships no bundled libs — the binary's NEEDED list is covered by
# device system libs (libSDL2, libGLESv2, libcurl). Verify and fail loudly
# if a future upstream bump adds a new dep we'd have to start shipping.
mkdir -p Oxygen/sonic3air/build/_cmake/build/libs
needed=$(readelf -d Oxygen/sonic3air/sonic3air_linux | awk -F'[][]' '/NEEDED/ {print $2}')
echo "Binary NEEDED:"; echo "$needed"
# Libs that are OK to not ship (device system provides them or they're stdlibs).
# ld-linux-aarch64.so.1 is the dynamic loader itself — part of glibc, always
# present on aarch64 systems, referenced via DT_NEEDED on some toolchains.
system_libs="libSDL2-2.0.so.0 libGLESv2.so.2 libcurl.so.4 \
             libc.so.6 libm.so.6 libpthread.so.0 libdl.so.2 librt.so.1 \
             libstdc++.so.6 libgcc_s.so.1 \
             ld-linux-aarch64.so.1"
while read -r lib; do
  [[ -z "$lib" ]] && continue
  if ! echo " $system_libs " | grep -q " $lib "; then
    echo "::warning::Binary NEEDs $lib which is not in the system-libs allowlist — consider shipping it explicitly"
  fi
done <<< "$needed"

cd ..
