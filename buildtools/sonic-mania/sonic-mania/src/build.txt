# GitHub occasionally returns HTTP 500s during git clone. Retry with backoff
# so transient failures don't abort the whole build. Use this instead of
# `git clone` directly for any remote clone. Derives the target dir from
# the URL basename (same as `git clone` itself) to clean up partial clones
# between attempts.
git_clone_retry() {
    local url target attempt
    # Find the URL arg — it's the last non-flag positional. If an explicit
    # target dir was passed after the URL, use that instead.
    for arg in "$@"; do
        case "$arg" in
            http*|git@*|ssh://*|git://*) url="$arg" ;;
            -*) ;;
            *) [[ -n "$url" ]] && target="$arg" ;;
        esac
    done
    if [[ -z "$target" && -n "$url" ]]; then
        target="${url##*/}"
        target="${target%.git}"
    fi
    for attempt in 1 2 3 4 5; do
        if git clone "$@"; then
            return 0
        fi
        # Clean up any partially-cloned dir so the next attempt can re-use
        # the target name.
        [[ -n "$target" && -d "$target" ]] && rm -rf "$target"
        local delay=$((attempt * 5))
        echo "git clone failed (attempt $attempt); retrying in ${delay}s..." >&2
        sleep "$delay"
    done
    echo "git clone failed after 5 attempts" >&2
    return 1
}

# We build SDL2 2.32 from source so the libs we ship later (none in Mania's
# case, but keeping the pattern) can link against a fast modern SDL2 at build
# time. We do NOT ship libSDL2-2.0.so.0 — devices provide their own, often
# with device-specific patches (GL backend selection, touchscreen quirks).
# Shipping ours would override the patched system copy via LD_LIBRARY_PATH
# and could regress things that currently work.
git_clone_retry https://github.com/libsdl-org/SDL.git
cd SDL
git checkout release-2.32.0
mkdir -p build-sonicmania && cd build-sonicmania
cmake .. -DBUILD_SHARED_LIBS=ON
make -j$(nproc)
make install
cd ../..

#### Sonic Mania Decompilation ####

git_clone_retry --recursive https://github.com/RSDKModding/Sonic-Mania-Decompilation
cd Sonic-Mania-Decompilation

# Pin to current HEAD; the upstream has no release tags
git log -1 --format='%H' > ../sonicmania.sha

cmake -S . -B build \
  -DCMAKE_BUILD_TYPE=Release \
  -DCMAKE_INTERPROCEDURAL_OPTIMIZATION=ON \
  -DRETRO_REVISION=2 \
  -DRETRO_SUBSYSTEM=SDL2 \
  -DRETRO_DISABLE_PLUS=ON

cmake --build build --config release -j$(nproc)

# Stage shipped libraries. Filenames MUST match the binary's DT_NEEDED entries
# exactly (verify with `readelf -d <binary> | grep NEEDED`). We use `cp -L` to
# follow symlinks so the destination is a real file with the right name —
# otherwise `cp` may preserve a symlink that points to a file not in the zip,
# producing a broken link, and the dynamic loader silently falls through to
# the system library. Don't use wildcards here: multiple matching files
# (SONAME symlink + versioned real file) cause duplicates or deleted targets.
mkdir -p build/libs
cp -L /usr/lib/aarch64-linux-gnu/libogg.so.0 ./build/libs/libogg.so.0
cp -L /usr/lib/aarch64-linux-gnu/libtheoradec.so.1 ./build/libs/libtheoradec.so.1

# Verify: every staged lib must be in the binary's NEEDED list, and every
# NEEDED lib we intend to ship must be staged. System libs (libc, libGLESv2,
# libSDL2, etc.) are assumed present on the device and are NOT shipped.
needed=$(readelf -d build/dependencies/RSDKv5/RSDKv5 | awk -F'[][]' '/NEEDED/ {print $2}')
echo "Binary NEEDED:"; echo "$needed"
echo "Staged:"; ls build/libs/
for lib in libogg.so.0 libtheoradec.so.1; do
    if ! echo "$needed" | grep -qx "$lib"; then
        echo "ERROR: staged $lib but binary does not NEED it"; exit 1
    fi
    if [ ! -f "build/libs/$lib" ]; then
        echo "ERROR: $lib missing from build/libs/"; exit 1
    fi
done

cd ..
