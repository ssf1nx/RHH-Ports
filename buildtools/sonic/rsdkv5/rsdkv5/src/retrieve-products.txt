#!/bin/bash
set -e

SRCDIR=$1
DESTDIR=$2

cd "$DESTDIR" || { echo "DESTDIR not found: $DESTDIR"; exit 1; }

cp "$SRCDIR/Sonic-Mania-Decompilation/build/dependencies/RSDKv5/RSDKv5" "$DESTDIR/sonicmania"
cp "$SRCDIR/Sonic-Mania-Decompilation/build/libGame.so" "$DESTDIR/Game.so"

rm -rf ./libs
cp -r "$SRCDIR/Sonic-Mania-Decompilation/build/libs" .

echo "Packaging complete."
