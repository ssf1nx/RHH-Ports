set -e
source /root/buildtools/harbourmasters/lib_build.sh

project_clone https://github.com/ssf1nx/2ship2harkinian.git

# See SoH build.txt for the rationale on building SDL2 from source but not
# shipping libSDL2 to devices.
build_sdl2

build_libzip
build_json
build_bzip2
build_tinyxml2
build_opus
build_opusfile

# 2s2h enables stricter warnings than SoH — relax a few to keep the build
# green against upstream churn.
export CFLAGS="-O3 -Wall -Wextra -Wno-return-type -Wno-unused-parameter -Wno-unused-function -Wno-unused-variable -Wno-macro-redefined -Wno-unknown-warning-option"
export CXXFLAGS="$CFLAGS"

project_configure_and_build Generate2ShipOtr \
    -DUSE_OPENGLES=1 \
    -DBUILD_CROWD_CONTROL=0 \
    -DCMAKE_BUILD_TYPE=${BUILD_TYPE:-Release}

# 2s2h adds spdlog to SoH's NEEDED set.
stage_libs mm/2s2h.elf \
    libz.so.1 \
    libogg.so.0 \
    libpng16.so.16 \
    libvorbis.so.0 \
    libvorbisenc.so.2 \
    libvorbisfile.so.3 \
    libspdlog.so.1 \
    libbz2.so.1 \
    libzip.so.5 \
    libtinyxml2.so.10 \
    libopus.so.0 \
    libopusfile.so.0
