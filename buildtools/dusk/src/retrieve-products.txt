#!/bin/bash
set -e

SRCDIR=$1
DESTDIR=$2

cd "$DESTDIR" || { echo "DESTDIR not found: $DESTDIR"; exit 1; }

# build.txt stages each artifact as a SIBLING of the dusk checkout — the
# dusk binary at $SRCDIR/dusk, libs at $SRCDIR/libs/, res at $SRCDIR/res/.
# This avoids colliding with directories of the same name inside the dusk
# source tree (notably $SRCDIR/dusk/libs/ which contains JSystem/PPC EABI
# headers that aren't shipped).

# Binary
cp "$SRCDIR/dusk" "$DESTDIR/dusk"
chmod +x "$DESTDIR/dusk"

# Bundled libs (currently just libfreetype). Wipe and replace so a future
# build that drops or adds a lib stays in sync with the staging dir.
rm -rf "$DESTDIR/libs"
mkdir -p "$DESTDIR/libs"
cp -a "$SRCDIR/libs/." "$DESTDIR/libs/"

# Upstream res/ — fonts, icons, and RML stylesheets that Dusk loads via
# SDL_GetBasePath() + "res/". Wipe-and-replace so dropped/renamed assets
# from upstream don't linger.
rm -rf "$DESTDIR/res"
cp -a "$SRCDIR/res" "$DESTDIR/res"

# Other files in the port dir (Dusk.sh, port.json, README.md, BUILDING.md,
# dusk.gptk, gameinfo.xml, baseroms/, config/) are RHH packaging or runtime
# state and are intentionally NOT synced — the workflow only touches paths
# listed in registry.json's `artifacts` field.

echo "Packaging complete."
